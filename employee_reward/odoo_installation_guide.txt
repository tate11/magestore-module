Step 1: Clone or download odoo from github
Link: https://github.com/odoo/odoo

Step 2: Installing postgres
+ sudo sh -c 'echo "deb http://apt.postgresql.org/pub/repos/apt/ `lsb_release -cs`-pgdg main" >> /etc/apt/sources.list.d/pgdg.list'
+ wget -q https://www.postgresql.org/media/keys/ACCC4CF8.asc -O - | sudo apt-key add -
+ sudo apt-get update
+ sudo apt-get install postgresql

Step 3: Configurating postgres
+ sudo su - postgres
+ psql
+ create role "your_system_username" superuser;
+ alter role "your_system_username" with login;
+ create database "your_system_username";
+ \q
+ exit

Step 4: Preparing odoo
+ sudo apt-get install python-pip
+ pip install upgrade pip
+ sudo apt-get install postgresql-server-dev-X.Y (postgres current version)
+ sudo apt-get install libxml2-dev libxslt1-dev libsasl2-dev python-dev libldap2-dev libssl-dev
+ pip install requirement.txt --user
+ sudo apt-get install npm
+ sudo ln -s /usr/bin/nodejs /usr/bin/node
+ sudo npm install -g less
